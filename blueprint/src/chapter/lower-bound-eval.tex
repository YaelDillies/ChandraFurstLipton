\chapter{Lower bound on the communication complexity of $\eval$}

\begin{definition}[$\eval$ function]
  \label{def:eval}
  % \lean{}
  % \leanok

  The $\eval$ function is defined as
  \begin{align}
      \eval : G^d & \to \Bool \\
      x & \mapsto \begin{cases}
      1 & \text{ if } \sum_i x_i = 0 \\
      0 & \text{ else}
    \end{cases}
  \end{align}
\end{definition}
